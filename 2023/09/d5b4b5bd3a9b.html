存储过程
创建
create procedure vvname(
IN pid int,
OUT pname char(50)
)
begin
select name
from user
where id = pid
into pname ;
end;
MySQL支持IN（传递给存储过程）、OUT（从存
储过程传出）和INOUT（对存储过程传入和传出）类型的参
数
使用
call vvname(1,@name);
select @name;
